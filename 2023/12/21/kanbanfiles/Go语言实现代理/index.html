用Go语言实现一个Web服务，处理用户的HTTP请求。

package main

import (
“fmt”
“log”
“net/http”
)

func main() {
http.HandleFunc(“/“, handler)
log.Fatal(http.ListenAndServe(“:8080”, nil))
}

func handler(w http.ResponseWriter, r *http.Request) {
fmt.Fprintf(w, “Hello, World!”)
}
